 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if (! head) return head;
        auto a = head, b = head -> next;
        while (b){
            auto c = b -> next;
            b -> next = a;
            a = b, b = c;
        }
        head -> next = NULL;
        head = a;
        return head;
    }

 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {

    ListNode* merge(ListNode* l1, ListNode* l2) {
        auto dummy = new ListNode(-1);
        auto cur = dummy;
        while (l1 != NULL && l2 != NULL){
            if (l1->val < l2->val){
                cur -> next = l1;
                l1 = l1 -> next;
            }
            else {
                cur -> next = l2;
                l2 = l2 -> next;
            }
            cur = cur -> next;
        }
        if (l1 == NULL) cur -> next = l2;
        else cur -> next = l1;
        return dummy -> next;
    }  
};
