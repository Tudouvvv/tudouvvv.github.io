class Solution:
    def projectionArea(self, grid: List[List[int]]) -> int:
        top, front, left = (0, 0, 0)
        for i in range(len(grid)):
            left_num = 0
            front += max(grid[i])
            for j in range(len(grid[0])):
                if grid[i][j] > 0:
                    top += 1
                left_num = max(left_num, grid[j][i])
            left += left_num
        return top + front + left

class Solution:
    def projectionArea(self, grid: 'List[List[int]]') -> 'int':
