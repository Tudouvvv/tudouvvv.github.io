class Solution:
    def Fibonacci_sequence(self, nums):
        f = [0 for _ in range(nums)]
        if nums == 1:
            return [1]
        if nums == 2:
            return [1, 1]
        f[0], f[1] = 1, 1
        for i in range(2, nums):
            f[i] = f[i - 1] + f[i - 2]
        return f

class Solution:
    def Prime(self, nums):
        f = [2]
        if nums == 1:
            return f
        i = 3
        while len(f) < nums:
            if self.is_prime(i): f.append(i)
            i += 1
        return f

    def is_prime(self, n):
        for i in range(2, int(n ** 0.5) + 1):
            if n % i == 0:
                return False
        return True
