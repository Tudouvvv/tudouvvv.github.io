class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        if (nums.empty()) return vector<int> {-1, -1};
        unordered_map<int, int> hash;
        for (int i = 0; i < nums.size(); i ++){
            if (hash.count(target - nums[i])) return vector<int> {hash[target - nums[i]], i};
            hash[nums[i]] = i;
        }
        return vector<int> {-1, -1};
    }
};

        if not nums: return [-1, -1]
        count = {}
        for i in range(len(nums)):
            if target - nums[i] in count: 
                return [count[target - nums[i]], i]
            count[nums[i]] = i
        return [-1, -1]

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        auto dummy = new ListNode(-1);
        auto cur = dummy;
        int carry = 0;
        while (l1 || l2){
            int n1 = l1 ? l1->val : 0;
            int n2 = l2 ? l2->val : 0;
            int sum = n1 + n2 + carry;
            carry = sum / 10;
            cur->next = new ListNode(sum % 10);
            cur = cur->next;
            if (l1) l1 = l1->next;
            if (l2) l2 = l2->next;
        }
        if (carry) cur->next = new ListNode(carry);
        return dummy->next;
    }
};

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        auto dummy = new ListNode(-1);
        auto cur = dummy;
        int carry = 0;
        while (l1 || l2 || carry){
            if (l1)
            {
                carry += l1->val;
                l1 = l1->next;
            }
            if (l2)
            {
                carry += l2->val;
                l2 = l2->next;
            }
            cur -> next = new ListNode(carry % 10);
            cur = cur->next;
            carry /= 10;
        }
        return dummy->next;
    }
};

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_map<char, int> hash;
        int res = 0;
        for (int i = 0, j = 0; i < s.size(); i++ ){
            hash[s[i]] ++;
            while (hash[s[i]] > 1) hash[s[j ++]] --;
            res = max(res, i - j + 1);
        }
        return res;
    }
};

class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        count = {}
        j = 0
        res = 0
        for i in range(len(s)):
            if s[i] in count: count[s[i]] += 1
            else: count[s[i]] = 1
            while (count[s[i]] > 1): 
                count[s[j]] -= 1
                j += 1
            res = max(res, i - j + 1)
        return res

class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        priority_queue<int, vector<int>, greater<int>> up;
        priority_queue<int> down;
        vector<int> nums;
        for (auto x : nums1) nums.push_back(x);
        for (auto x : nums2) nums.push_back(x);
        for (auto x : nums){
            if(down.empty() || x > down.top()) up.push(x);
            else{
                down.push(x);
                up.push(down.top());
                down.pop();
            }
            if (up.size() > down.size() + 1){
                down.push(up.top());
                up.pop();
            }
        }
        if ((down.size() + up.size()) % 2 == 1) return up.top();
        else return (down.top() + up.top()) * 0.5;
    }
};

class Solution {
public:
    string longestPalindrome(string s) {
        string res;
        for (int i = 0; i < s.size(); i ++){
            for (int j = i, k = i; j >= 0 && k < s.size() && s[j] == s[k]; j --, k ++){
                if (res.size() < (k - j + 1)) res = s.substr(j, k - j + 1);
            }
            for (int j = i, k = i + 1; j >= 0 && k <= s.size() && s[j] == s[k]; j --, k ++){
                if (res.size() < (k - j + 1)) res = s.substr(j, k - j + 1);
            }
        }
        return res;
    }
};

class Solution:
    def longestPalindrome(self, s: str) -> str:
        i = 0
        res = ""
        while i < len(s):
            j, k = i, i
            while (j >=0 and k < len(s) and s[j] == s[k]):
                if len(res) < k - j + 1:
                    res = s[j : k + 1]
                j -= 1
                k += 1
            p, q = i, i+1
            while (p >= 0 and q < len(s) and s[p] == s[q]):
                if len(res) < q - p + 1:
                    res = s[p : q + 1]
                p -= 1
                q += 1
            i += 1
        return res

class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> res;
        sort(nums.begin(), nums.end());
        for (int i = 0; i < nums.size(); i ++){
            if (i > 0 && i < nums.size() && nums[i] == nums[i - 1]) continue;
            int l = i + 1, r = nums.size() - 1;
            while (l < r){
                int s = nums[i] + nums[l] + nums[r];
                if (s > 0) r --;
                else if (s < 0) l ++;
                else{
                    res.push_back(vector<int>{nums[i], nums[l], nums[r]});
                    while(l < r && nums[l] == nums[l + 1]) l ++;
                    while(l < r && nums[r] == nums[r - 1]) r --;
                    l ++, r --;
                }
            }
        }
        return res;
    }
};

class Solution {
public:
    vector<string> letterCombinations(string digits) {
        if (digits.empty()) return vector<string> {};
        vector<string> states (1, "");
        string chars[8] = {"abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"};
        
        for (auto u : digits)
        {
            vector<string> res;
            for (auto c : chars[u - '2'])
                for (auto s : states)
                    res.push_back(s + c);
            states = res;
        }
        return states;
    }
};

class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        if not digits: return None;
        chars = ['abc', 'def', 'ghi', 'jkl', 'mno', 'pqrs',
                'tuv', 'wxyz']
        states = [""]
        for u in digits:
            res = []
            for c in chars[int(u) - 2]:
                for s in states:
                    res.append(s + c)
            states = res
        return states

 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        auto dummy = new ListNode(-1);
        dummy->next = head;
        auto first = dummy, second = dummy;
        while (n --) first = first->next;
        while(first->next){
            first = first -> next;
            second = second->next;
        }
        second->next = second->next->next;
        return dummy->next;
    }
};
