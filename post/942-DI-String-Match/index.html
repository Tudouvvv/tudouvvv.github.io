class Solution:
    def diStringMatch(self, S: str) -> List[int]:
        p = []
        N = []
        for i in range(len(S)+ 1):
            p.append(i)
        for char in S:
            if char == "D":
                N.append(max(p))
                p.remove(max(p))
            else:
                N.append(min(p))
                p.remove(min(p))
        N.append(p[0])
        return N

class Solution:
    def diStringMatch(self, S: str) -> List[int]:
        l, r, N = 0, len(S), []
        for char in S:
            N.append(l if char =='I' else r)
            l, r = l +(char == 'I'), r - (char == 'D')
        return N + [l]
